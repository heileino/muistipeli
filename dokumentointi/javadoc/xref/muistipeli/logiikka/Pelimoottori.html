3   /**
4    * Luokka toimii linkkinä käyttöliittymäluokkien ja muiden logiikka-luokkien
5    * välillä. Luokka tarjoaa metodeita, joilla peli etenee alusta loppuun.
6    *
7    * @author Heikki Leinonen
8    */
9   public class Pelimoottori {
10  
11      private final int korttienMaara;
12      private String parasTulosTiedostonNimi;
13      private Pelipoyta pelipoyta;
14      private LoytyneetKortit loytyneetKortit;
15      private ParhaanTuloksenTilasto parasTilasto;
16      private ValittujenIndeksienSailio valittujenIndeksienSailio;
17      private Yritysmaaralaskuri yrityslaskuri;
18  
19      /**
20       * Konstruktori luo tarvittavat tiedot ja apuvälineet pelin pelaamiseen.
21       */
22      public Pelimoottori() {
23          korttienMaara = 16;
24          parasTulosTiedostonNimi = "parastulos.txt";
25          pelipoyta = new Pelipoyta(korttienMaara);
26          loytyneetKortit = new LoytyneetKortit();
27          parasTilasto = new ParhaanTuloksenTilasto(parasTulosTiedostonNimi);
28          valittujenIndeksienSailio = new ValittujenIndeksienSailio();
29          yrityslaskuri = new Yritysmaaralaskuri();
30      }
31  
32      /**
33       * Metodi lisää taulukon indeksiä vastaavan kortin valittujen joukkoon ja
34       * kääntää sen asennoksi kuvapuolen, mikäli kortti on valintakelpoinen.
35       * Kortti on valintakelpoinen, mikäli sen asento on selkäpuoli.
36       *
37       * @param i korttitaulukon indeksi
38       *
39       * @return tosi jos valinta onnistuu
40       */
41      public boolean valitaanKorttiJosMahdollista(int i) {
42          if (pelipoyta.nakyykoKortinKuvapuoli(i)) {
43              return false;
44          }
45          valittujenIndeksienSailio.lisaaValittuihin(i);
46          pelipoyta.paljastaKortinKuva(i);
47          return true;
48      }
49  
50      /**
51       * Metodi tarkistaa pelitilanteen yrityskerran sisällä. Vaihtoehtoina ovat
52       * toisen kortin valinta tai jatkaminen korttien vertailuun.
53       *
54       * @return tosi, jos toisen kortin valinnalle on tarvetta
55       */
56      public boolean valitaankoToinenKortti() {
57          if (valittujenIndeksienSailio.montakoValittu() > 1) {
58              yrityslaskuri.lisaaValintayritys();
59              return false;
60          }
61          return true;
62      }
63  
64      /**
65       * Metodi välittää kutsujalle taulukon indeksiä vastaavan kortin tunnisteen.
66       *
67       * @param i taulukon indeksi
68       *
69       * @return kortin tunniste
70       */
71      public String getKortinTunniste(int i) {
72          return pelipoyta.getKorttiTaulukosta(i).toString();
73      }
74  
75      /**
76       * Metodi lisää valittujen korttien listalla olevat kortit löydettyjen
77       * korttien listalle, mikäli ne ovat pari tai kääntää korttien selkäpuolet
78       * esiin, mikäli valitut kortit eivät ole pareja.
79       *
80       * @return tosi, jos pari löytyy
81       */
82      public boolean ovatkoValinnatPareja() {
83          Kortti kortti1 = getKorttiValittujenKorttienJoukosta(0);
84          Kortti kortti2 = getKorttiValittujenKorttienJoukosta(1);
85          if (pelipoyta.onkoKorteillaSamaTunnus(kortti1, kortti2)) {
86              loytyneetKortit.lisaaKortitLoytyneeksi(kortti1, kortti2);
87              return true;
88          }
89          pelipoyta.piilotaKortinKuva(pelipoyta.getKortinIndeksi(kortti1));
90          pelipoyta.piilotaKortinKuva(pelipoyta.getKortinIndeksi(kortti2));
91          return false;
92      }
93  
94      /**
95       * Metodi päättää yhden yrityskerran nollaamalla valitut taulukkoindeksit ja
96       * selvittää, jatketaanko peliä vielä päättyneen yrityksen jälkeen. Peli
97       * jatkuu, mikäli korttipareja on vielä löytymättä.
98       *
99       * @return tosi, jos peli jatkuu
100      */
101     public boolean jatkuukoPeli() {
102         valittujenIndeksienSailio = new ValittujenIndeksienSailio();
103         return getLoytymattomienParienLukumaara() > 0;
104     }
105 
106     /**
107      * Metodi täyttää pelipöydän korteilla, sekoittaa ne satunnaiseen
108      * järjestykseen ja tähän asti saadun parhaan tuloksen lataaminen ohjelmaan.
109      */
110     public void pelaaPeli() {
111         pelipoyta.asetaKortitTaulukkoon();
112         pelipoyta.sekoitaTaulukonKortit();
113         parasTilasto.lataaParasTulos();
114     }
115 
116     /**
117      * Metodi tarkastaa onko pelattu tulos paras. Jos tulos on parempi kuin
118      * voimassa oleva paras tulos, metodi korvaa vanhan tuloksen uudella.
119      *
120      * @return tosi, jos pelin tulos on parempi kuin voimassa oleva paras tulos
121      */
122     public boolean onUusiParasTulos() {
123         if (yrityslaskuri.getYritysmaara() < parasTilasto.getParasTulos() || parasTilasto.getParasTulos() == 0) {
124             parasTilasto.setParasTulos(yrityslaskuri.getYritysmaara());
125             parasTilasto.lataaParasTulos();
126             return true;
127         }
128         return false;
129     }
130 
131     /**
132      * Metodi joko käynnistää tai lopettaa pelin.
133      *
134      * @param jatkohalu käyttäjän toivettava vastaava numero
135      *
136      * @return tosi, jos parametri on 0
137      */
138     public boolean aloitetaankoUusiPeli(int jatkohalu) {
139         return jatkohalu == 0;
140     }
141 
142     /**
143      * Metodi laskee ja palauttaa löytämättä olevien korttien lukumäärän
144      *
145      * @return löytämättömien korttien lukumäärä
146      */
147     public int getLoytymattomienParienLukumaara() {
148         return (korttienMaara - getLoytyneet().getLoydetytKortit().size()) / 2;
149     }
150 
151     public Kortti getKorttiValittujenKorttienJoukosta(int valinnanIndeksi) {
152         return this.pelipoyta.getKorttiTaulukosta(this.valittujenIndeksienSailio.getValitutIndeksit().get(valinnanIndeksi));
153     }
